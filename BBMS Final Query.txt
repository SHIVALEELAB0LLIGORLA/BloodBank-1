SqlConnection conn = new SqlConnection(ConfigurationManager.ConnectionStrings["bbEntities"].ConnectionString);
        SqlCommand cmd = null;
        SqlDataAdapter da = null;
        [HttpPost]
        [Route("Registration")]
        public string Registration(DonorLogin donor)
        {
            string msg = string.Empty;
            try
            {
                da = new SqlDataAdapter("Donor_Registration", conn);
                da.SelectCommand.CommandType = CommandType.StoredProcedure;
                da.SelectCommand.Parameters.AddWithValue("@Name", donor.Name);
                da.SelectCommand.Parameters.AddWithValue("@Password", donor.Password);
                da.SelectCommand.Parameters.AddWithValue("@Age", donor.Age);
                da.SelectCommand.Parameters.AddWithValue("@Gender", donor.Gender);
                da.SelectCommand.Parameters.AddWithValue("@Email", donor.Email);
                da.SelectCommand.Parameters.AddWithValue("@Mobile", donor.Mobile);
                da.SelectCommand.Parameters.AddWithValue("@Address", donor.Address);


                SqlParameter outParam = new SqlParameter("@IsExist", SqlDbType.Int);
                outParam.Direction = ParameterDirection.Output;
                da.SelectCommand.Parameters.Add(outParam);

                conn.Open();
                da.SelectCommand.ExecuteNonQuery();
                conn.Close();

                int isExist = Convert.ToInt32(da.SelectCommand.Parameters["@IsExist"].Value);
                if (isExist == 1)
                {
                    msg = "User already exists.";
                }
                else
                {
                    msg = "Registration successful!";
                }

            }
            catch (Exception ex)
            {
                msg = ex.Message;
            }
            return msg;
        }
